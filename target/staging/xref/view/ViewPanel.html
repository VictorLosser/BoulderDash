18  
19  	
20  	
21  	private static final long	serialVersionUID	= -998294702363713521L;
22  	
23  	BufferedImage imgHero;
24      Timer timer;
25      int i;
26      Image subSpriteHero;
27      
28      int[][] spriteSheetCoordsWalkLeft = { { 16, 16, 16, 16 }, { 0, 16, 16, 16 }, { 16, 16, 16, 16 }, { 0, 16, 16, 16 }};
29  
30      public ViewPanel() {
31          setBackground(Color.black);
32          try
33          {
34              imgHero = ImageIO.read(new File("D:/Document/test projects/Sprites/Rockford.png"));
35              timer = new Timer();
36              timer.scheduleAtFixedRate(new TimerTask() {
37                  @Override
38                  public void run() {
39                      subSpriteHero = imgHero.getSubimage(spriteSheetCoordsWalkLeft[i][0], spriteSheetCoordsWalkLeft[i][1], spriteSheetCoordsWalkLeft[i][2],
40                      		spriteSheetCoordsWalkLeft[i][3]);
41                      i++;
42                      if (i == spriteSheetCoordsWalkLeft.length) {
43                          i = 0;
44                      }
45                      
46                      
47                      repaint();
48                      revalidate();
49                  }
50              }, 350, 350);
51          }
52          catch (IOException e)
53          {
54              e.printStackTrace();
55          }
56          
57      }
58  
59      @Override
60      public void paintComponent(Graphics g) {
61          super.paintComponent(g);
62          g.drawImage(subSpriteHero, 0, 0, 32, 32, null);
63        
64      }
