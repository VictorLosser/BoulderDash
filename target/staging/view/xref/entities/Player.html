18          super(handler, x, y, Creature.DEFAULT_CREATURE_HEIGHT, Creature.DEFAULT_CREATURE_WIDTH);

36      public void tick() {
37      	down.tock();
38      	up.tock();
39      	left.tock();
40      	right.tock();
41      	staticP.tock();
42          getInput();
43          move();
44          handler.getGameCamera().centerOnEntity(this);
45          checkTouch();
46      }

48      private void checkTouch(){
49      	Rectangle cb = getCollisionBounds(0,0);
50      	Rectangle ar = new Rectangle();
51      	Rectangle ar2 = new Rectangle();
52      	
53      	int arSize = 8;
54      	int arSize2 = 32;
55      	
56      	ar.width = arSize;
57      	ar.height = arSize;
58      	ar2.width = arSize2;
59      	ar2.height = arSize2;
60      	
61      	if(handler.getKeyManager().dup){
62      		ar2.x = cb.x+ cb.width / 2 - arSize2 / 2;
63      		ar2.y = cb.y - arSize2;
64      	}else if(handler.getKeyManager().ddown){
65          		ar2.x = cb.x + cb.width / 2 -arSize2 / 2;
66          		ar2.y = cb.y + cb.height;
67      	}else if(handler.getKeyManager().dleft){
68              		ar2.x = cb.x - arSize2;
69              		ar2.y = cb.y+ cb.height / 2 -arSize2 /2;
70          		
71          }else if(handler.getKeyManager().dright){
72                  		ar2.x = cb.x + cb.width;
73                  		ar2.y = cb.y + cb.height / 2 -arSize2 /2 ;
74        
75          }else if(handler.getKeyManager().up){
76      		ar.x = cb.x+ cb.width / 2 - arSize / 2;
77      		ar.y = cb.y - arSize;
78      	}else if(handler.getKeyManager().down){
79          		ar.x = cb.x + cb.width / 2 -arSize / 2;
80          		ar.y = cb.y + cb.height;
81      	}else if(handler.getKeyManager().left){
82              		ar.x = cb.x - arSize;
83              		ar.y = cb.y+ cb.height / 2 -arSize /2;
84          }else if(handler.getKeyManager().right){
85                  		ar.x = cb.x + cb.width;
86                  		ar.y = cb.y + cb.height / 2 -arSize /2 ;    		
87          }else{
88          	return;
89          }
90      	
91      		for(Entity e : handler.getWorld().getEntityManager().getEntities()){
92      			if(e.equals(this))
93      				continue;
94      			
95      			if(e.getCollisionBounds(0, 0).intersects(ar)){
96      				e.touch(1);
97      				return;
98      			}
99      			if(e.getCollisionBounds(0, 0).intersects(ar2)){
100     				e.touch(1);
101     				return;
102     			}
103     		
104     		for(Entity i : handler.getWorld().getEntityManager().getItem()){
105     			if(i.equals(this))
106     				continue;
107     			if(i.getCollisionBounds(0, 0).intersects((int) (x + bounds.x), (int) (y + bounds.y), bounds.width, bounds.height)){
108     				i.touch(1);
109     				//this.score.setNumber_diamond(this.score.getNumber_diamond() + 1);
110     				return;
111     			}		
112     			}
113     	}
114     }

123     private void getInput(){
124         xMove = 0;
125         yMove = 0;
126 
127         //Here we apply the movement corresponding to the keyboard input
128         if(handler.getKeyManager().up)
129             yMove = -speed;
130         if(handler.getKeyManager().down)
131             yMove = speed;
132         if(handler.getKeyManager().left)
133             xMove = -speed;
134         if(handler.getKeyManager().right)
135             xMove = speed;
136     }

150     public BufferedImage getCurrentFrame(){ //This method return the image corresponding to the movement
151     	if(xMove < 0){
152     		return left.getCurrentFrame();
153     		
154     	}else if(xMove > 0){
155     		return right.getCurrentFrame();
156     		
157     	}else if(yMove < 0){
158     		return up.getCurrentFrame();
159     		
160     	}else if(yMove > 0){
161     		return down.getCurrentFrame();
162     	}else{
163     		return staticP.getCurrentFrame();
